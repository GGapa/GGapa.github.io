T3 100pts\n


T3 我依然从特殊性质开始打起，这也从侧面说明了特殊性质的重要性，有时候完全可以通过特殊性质来发现正解。在考试的时候并没有证明结论的正确性，因为是IOI赛制的缘故提交并且正确了。\n

结论就是找离 $n$ 最近的质数，不是离 $n$ 最近的质数就全部取 1，后判断一下大小。\n

\n#include <iostream>\n#include <cmath>\n#include <cstdio>\n#include <bitset>\nusing namespace std;\ntypedef long long ll;\n\nll T, n, k;\n\nll qpow(ll x, ll y) {\n    ll ans = 1, base = x;\n    while(y != 0) {\n        if(y % 2) ans = (ans * base);\n        base = (base * base);\n        y >>= 1;\n    }\n    return ans;\n}\n\nll qmul(ll x, ll y) {\n    ll ans = 0;\n    while(y != 0) {\n        if(y % 2) ans = (ans + x);\n        x = (x + x);\n        y >>= 1;\n    }\n    return ans;\n}\n\nbool check(ll x) {\n    if(x == 2 || x == 1) return true;\n    int si = sqrt(x) + 1;\n    for(int i = 2; i <= si; i++) {\n        if(x % i == 0) return false;\n    }\n    return true;\n}\n\nll sol(ll x) {\n    if(x <= 0) return 0;\n    ll tmp = x;\n    while(!check(x)) x--;\n    return max(sol(tmp - x) + qpow(x-k, 2), qmul(qpow(1 - k, 2), tmp));\n}\n\nint main() {\n    ios::sync_with_stdio(0);\n    cin.tie(0);\n    cout.tie(0);\n    \/\/ freopen("AKIOI.in", "r", stdin);\n    \/\/ freopen("AKIOI.out", "w", stdout);\n    cin >> T;\n    while(T--) {\n        cin >> n >> k;\n        cout << sol(n) << "\\n";\n    }\n    return 0;\n}\n
